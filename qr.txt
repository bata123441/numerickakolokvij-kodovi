A = np.array ([[2, 1],
               [4, 1],
               [5, 1],
               [6, 1]])

b = np.array ([[100],
               [140],
               [120],
               [90]])

Q, R = np.linalg.qr(A)

x, _, _, _ = np.linalg.lstsq(R, Q.T @ b, rcond = None)
print(x)

beta, alpha = x[1, 0], x[0, 0]
print("beta =", beta, ", alpha =", alpha)
